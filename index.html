var fBuild = _getOrCreateFolder(visitFolder, '03_المباني_والمرافق'); var fWorkshop = _getOrCreateFolder(visitFolder, '04_الورش_والمعامل'); var fEquip = _getOrCreateFolder(visitFolder, '05_المعدات_والتجهيزات'); dnp._savedServer = new Date().toISOString(); var dataFolder = _getOrCreateFolder(root, DATA_FOLDER); // اسم ثابت لكل مركز ليُحدَّث بدل التكرار var fname = (data.code || 'مركز') + '.json'; var existing = dataFolder.getFilesByName(fname); while (existing.hasNext()) existing.next().setTrashed(true); // احذف القديم dataFolder.createFile(Utilities.newBlob(JSON.stringify(dnp), 'application/json', fname)); // نسخة داخل مجلد الزيارة أيضاً visitFolder.createFile(Utilities.newBlob(JSON.stringify(dnp, null, 2), 'application/json', 'بيانات_' + (data.code || 'مركز') + '.json')); } } // ====== قراءة مركزية: GET ====== // ?action=list → قائمة بكل الزيارات المحفوظة (ملخّص) // ?action=get&code=C01 → بيانات زيارة مركز محدّد كاملة function doGet(e) { var callback = (e && e.parameter && e.parameter.callback) || ''; try { var action = (e && e.parameter && e.parameter.action) || ''; var root = DriveApp.getFolderById(ROOT_FOLDER_ID); var dataFolder = _getOrCreateFolder(root, DATA_FOLDER); if (action === 'list') { var out = []; var files = dataFolder.getFiles(); while (files.hasNext()) { var f = files.next(); if (f.getName().slice(-5) !== '.json') continue; try { var d = JSON.parse(f.getBlob().getDataAsString('UTF-8')); out.push({ code: d.code || '', name: d.name || '', gov: d.gov || '', date: d.date || '', surveyor: d.surveyor || '', savedAt: d.savedAt || '', score: (d.notes || {}).score || '', cls: (d.notes || {}).cls || '', savedServer: d._savedServer || '', folder: d._folder || '', folderUrl: d._folder || '' }); } catch (err) {} } return _out({ ok: true, visits: out }, callback); } if (action === 'get') { var code = (e.parameter.code || '').trim(); var ff = dataFolder.getFilesByName(code + '.json'); if (ff.hasNext()) return _out({ ok: true, visit: JSON.parse(ff.next().getBlob().getDataAsString('UTF-8')) }, callback); return _out({ ok: false, error: 'لا توجد زيارة لهذا المركز' }, callback); } return _out({ ok: true, message: 'خدمة الزيارات الميدانية تعمل ✓' }, callback); } catch (err) { return _out({ ok: false, error: String(err) }, callback); } } // مخرَج يدعم JSONP (callback) لتجاوز قيود CORS عند الاستضافة على GitHub function _out(obj, callback) { var json = JSON.stringify(obj); if (callback) { return ContentService.createTextOutput(callback + '(' + json + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); } function _sendEmail(data, uploaded, folderUrl) { try { var to = NOTIFY_EMAIL || Session.getEffectiveUser().getEmail(); ' + '

'; MailApp.sendEmail({ to: to, subject: subject, htmlBody: html }); } catch (err) {} } function _getOrCreateFolder(parent, name) { name = _safe(name); var it = parent.getFoldersByName(name); return it.hasNext() ? it.next() : parent.createFolder(name); } function _safe(s) { return String(s || 'بدون_اسم').replace(/[\\\/:*?"<>|]/g, '_').substring(0, 100).trim() || 'بدون_اسم'; } function _dataUrlToBlob(dataUrl, name) { var m = dataUrl.match(/^data:([^;]+);base64,(.*)$/); var mime = m ? m[1] : 'image/jpeg'; var b64 = m ? m[2] : dataUrl; var ext = mime.indexOf('png') > -1 ? '.png' : '.jpg'; if (!/\.(jpg|jpeg|png|webp)$/i.test(name)) name = name + ext; return Utilities.newBlob(Utilities.base64Decode(b64), mime, _safe(name)); } function _today() { return Utilities.formatDate(new Date(), 'GMT+2', 'yyyy-MM-dd'); } function _json(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function _logToSheet(data, uploaded, folderUrl) { try { var root = DriveApp.getFolderById(ROOT_FOLDER_ID); var ss; var files = root.getFilesByName(SHEET_NAME); if (files.hasNext()) ss = SpreadsheetApp.open(files.next()); else { ss = SpreadsheetApp.create(SHEET_NAME); var f = DriveApp.getFileById(ss.getId()); root.addFile(f); DriveApp.getRootFolder().removeFile(f); } var sh = ss.getSheets()[0]; if (sh.getLastRow() === 0) { sh.appendRow(['تاريخ التسجيل','كود','اسم المركز','المحافظة','القائم بالزيارة','تاريخ الزيارة','الطاقة','عدد المباني','عدد الورش','العاملون','التقييم %','التصنيف','الصور','رابط المجلد']); sh.setFrozenRows(1); } var ws = 0; (data.facilities || []).forEach(function (f) { if (f.type === 'ورش' || f.type === 'معامل') ws++; }); sh.appendRow([new Date(), data.code || '', data.name || '', data.gov || '', data.surveyor || '', data.date || '', data.capacity || '', (data.buildings || []).length, ws, (data.staff || {}).total || 0, (data.notes || {}).score || '', (data.notes || {}).cls || '', uploaded, folderUrl]); } catch (err) {} }
